async function redirecionar() {
  const ref = db.collection("sistema").doc("contador");

  await db.runTransaction(async (transaction) => {
    const doc = await transaction.get(ref);
    let valor = doc.data().valor;

    valor++;

    transaction.update(ref, { valor: valor });

    let indice = valor % numeros.length;

    let numero = numeros[indice];

    window.location.href =
      `https://wa.me/${numero}?text=Olá vim do Instagram`;
  });
}
